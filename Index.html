        // Function to simulate fetching data from Gemini API
        async function getAiGeneratedPlan(prompt) {
             const apiKey = ""; // Leave empty, will be handled by the environment
             const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

            const payload = {
                contents: [{
                    parts: [{
                        text: prompt
                    }]
                }],
            };

            const response = await fetch(apiUrl, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(payload)
            });

            if (!response.ok) {
                throw new Error(`API call failed with status: ${response.status}`);
            }
            
            const result = await response.json();
            const candidate = result.candidates?.[0];

            progressData.labels = combined.map(item => item.label);
            progressData.weights = combined.map(item => item.weight);
